# Mini-Controller System Implementation

This content file provides the implementation of the Multi-Node Motor Controller system following the spec-driven approach. All implementation follows the system specification created in the previous phase, demonstrating how to integrate all learned concepts in a cohesive project that showcases practical robotics development.

## Implementation Overview

The Mini-Controller System implementation realizes the specification created in the previous document. This implementation demonstrates how to bring together all the concepts learned in the 4-layer pedagogical framework into a functioning multi-node system.

### Implementation Phases

The implementation follows the phased approach outlined in the specification:

1. **Phase 1**: Basic node skeletons with interface definitions
2. **Phase 2**: Core functionality with basic algorithms
3. **Phase 3**: Safety features and error handling
4. **Phase 4**: System integration and validation
5. **Phase 5**: Performance optimization and documentation

## Phase 1: Basic Node Skeletons

### 1.1 Sensor Node Implementation

First, let's implement a basic sensor node skeleton that follows the specification:

```python
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import LaserScan, Imu, JointState
from std_msgs.msg import Float32
import random
import math


class BasicSensorNode(Node):
    """
    Basic sensor node implementation following the specification
    """
    def __init__(self, node_name='basic_sensor', sensor_type='simulated'):
        super().__init__(node_name)

        # Declare parameters as per specification
        self.declare_parameter('publish_rate', 10.0)
        self.declare_parameter('frame_id', f'{sensor_type}_frame')
        self.declare_parameter('sensor_range_min', 0.1)
        self.declare_parameter('sensor_range_max', 30.0)

        # Get parameter values
        self.publish_rate = self.get_parameter('publish_rate').value
        self.frame_id = self.get_parameter('frame_id').value
        self.range_min = self.get_parameter('sensor_range_min').value
        self.range_max = self.get_parameter('sensor_range_max').value

        # Create publisher based on sensor type
        if sensor_type == 'lidar':
            self.publisher = self.create_publisher(LaserScan, 'sensor1_data', 10)
        elif sensor_type == 'imu':
            self.publisher = self.create_publisher(Imu, 'sensor2_data', 10)
        elif sensor_type == 'joint':
            self.publisher = self.create_publisher(JointState, 'sensor3_data', 10)
        else:
            self.publisher = self.create_publisher(Float32, 'sensor_generic', 10)

        # Create timer for publishing
        self.timer = self.create_timer(1.0/self.publish_rate, self.publish_sensor_data)

        self.get_logger().info(f'{node_name} initialized with {sensor_type} sensor')

    def publish_sensor_data(self):
        """
        Publish simulated sensor data based on sensor type
        """
        sensor_type = self.get_parameter('frame_id').value.split('_')[0]  # Extract type from frame_id

        if sensor_type == 'lidar':
            msg = self.create_lidar_scan()
            self.publisher.publish(msg)
        elif sensor_type == 'imu':
            msg = self.create_imu_data()
            self.publisher.publish(msg)
        elif sensor_type == 'joint':
            msg = self.create_joint_state()
            self.publisher.publish(msg)
        else:
            # Generic sensor data
            msg = Float32()
            msg.data = random.random()
            self.publisher.publish(msg)

    def create_lidar_scan(self):
        """
        Create simulated laser scan data
        """
        msg = LaserScan()
        msg.header.stamp = self.get_clock().now().to_msg()
        msg.header.frame_id = self.frame_id

        # Set scan parameters
        msg.angle_min = -math.pi / 2
        msg.angle_max = math.pi / 2
        msg.angle_increment = math.pi / 180  # 1 degree increments
        msg.time_increment = 0.0
        msg.scan_time = 1.0 / self.publish_rate
        msg.range_min = self.range_min
        msg.range_max = self.range_max

        # Create ranges (simulated obstacle at 2m in front)
        num_ranges = int((msg.angle_max - msg.angle_min) / msg.angle_increment) + 1
        msg.ranges = [random.uniform(1.0, 10.0) for _ in range(num_ranges)]
        # Add an obstacle in front
        center_idx = num_ranges // 2
        msg.ranges[center_idx] = 2.0  # Obstacle 2 meters away

        msg.intensities = [100.0 for _ in range(num_ranges)]  # Simulated intensities

        return msg

    def create_imu_data(self):
        """
        Create simulated IMU data
        """
        msg = Imu()
        msg.header.stamp = self.get_clock().now().to_msg()
        msg.header.frame_id = self.frame_id

        # Simulate some movement
        import time
        t = time.time()

        # Orientation (simulated rotation)
        msg.orientation.x = 0.0
        msg.orientation.y = 0.0
        msg.orientation.z = math.sin(t / 2.0)
        msg.orientation.w = math.cos(t / 2.0)

        # Angular velocity (rotating at 1 rad/s)
        msg.angular_velocity.x = 0.1
        msg.angular_velocity.y = 0.1
        msg.angular_velocity.z = 1.0

        # Linear acceleration (gravity + motion)
        msg.linear_acceleration.x = math.sin(t)
        msg.linear_acceleration.y = math.cos(t)
        msg.linear_acceleration.z = 9.81  # Gravity

        return msg

    def create_joint_state(self):
        """
        Create simulated joint state data
        """
        msg = JointState()
        msg.header.stamp = self.get_clock().now().to_msg()
        msg.header.frame_id = self.frame_id

        # Define some joints
        msg.name = ['joint1', 'joint2', 'joint3']
        msg.position = [math.sin(time.time()), math.cos(time.time()), 0.5 * math.sin(2 * time.time())]
        msg.velocity = [math.cos(time.time()), -math.sin(time.time()), math.cos(2 * time.time())]
        msg.effort = [0.0, 0.0, 0.0]

        return msg


def main(args=None):
    rclpy.init(args=args)

    # Create sensor nodes based on type
    lidar_node = BasicSensorNode('lidar_sensor', 'lidar')
    imu_node = BasicSensorNode('imu_sensor', 'imu')
    joint_node = BasicSensorNode('joint_sensor', 'joint')

    try:
        # Run all nodes
        executor = rclpy.executors.MultiThreadedExecutor()
        executor.add_node(lidar_node)
        executor.add_node(imu_node)
        executor.add_node(joint_node)

        executor.spin()
    except KeyboardInterrupt:
        pass
    finally:
        lidar_node.destroy_node()
        imu_node.destroy_node()
        joint_node.destroy_node()
        rclpy.shutdown()


if __name__ == '__main__':
    main()
```

### 1.2 Controller Node Implementation

Now, let's implement the controller node that processes sensor data and generates motor commands:

```python
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import LaserScan, Imu, JointState
from geometry_msgs.msg import Twist
from std_msgs.msg import Float64, String
from std_srvs.srv import Trigger
import math
import time


class ControllerNode(Node):
    """
    Controller node that processes sensor data and generates motor commands
    """
    def __init__(self, node_name='controller_node'):
        super().__init__(node_name)

        # Declare parameters as per specification
        self.declare_parameter('control_frequency', 50.0)  # Hz
        self.declare_parameter('safety_timeout', 1.0)     # seconds
        self.declare_parameter('control_algorithm', 'obstacle_avoidance')
        self.declare_parameter('max_linear_velocity', 1.0)
        self.declare_parameter('max_angular_velocity', 1.0)

        # Get parameter values
        self.control_freq = self.get_parameter('control_frequency').value
        self.safety_timeout = self.get_parameter('safety_timeout').value
        self.control_algorithm = self.get_parameter('control_algorithm').value
        self.max_lin_vel = self.get_parameter('max_linear_velocity').value
        self.max_ang_vel = self.get_parameter('max_angular_velocity').value

        # Initialize sensor data storage
        self.lidar_data = None
        self.imu_data = None
        self.joint_data = None
        self.last_sensor_time = time.time()

        # Create subscribers for sensor data
        self.lidar_sub = self.create_subscription(
            LaserScan, 'sensor1_data', self.lidar_callback, 10)
        self.imu_sub = self.create_subscription(
            Imu, 'sensor2_data', self.imu_callback, 10)
        self.joint_sub = self.create_subscription(
            JointState, 'sensor3_data', self.joint_callback, 10)

        # Create publishers for motor commands
        self.cmd_pub = self.create_publisher(Twist, 'motor1_command', 10)
        self.motor1_pub = self.create_publisher(Float64, 'motor1_command', 10)
        self.motor2_pub = self.create_publisher(Float64, 'motor2_command', 10)

        # Create service servers
        self.emergency_stop_srv = self.create_service(
            Trigger, 'emergency_stop', self.emergency_stop_callback)
        self.reset_controller_srv = self.create_service(
            Trigger, 'reset_controller', self.reset_controller_callback)

        # Create timer for control loop
        self.control_timer = self.create_timer(
            1.0/self.control_freq, self.control_loop)

        # System state
        self.system_state = 'ACTIVE'  # ACTIVE, EMERGENCY_STOPPED, FAULT, INITIALIZING
        self.obstacle_detected = False
        self.obstacle_distance = float('inf')

        self.get_logger().info(f'{node_name} initialized with control frequency {self.control_freq}Hz')

    def lidar_callback(self, msg):
        """
        Handle incoming LIDAR data
        """
        self.lidar_data = msg
        self.last_sensor_time = time.time()

        # Process LIDAR data to detect obstacles
        if self.lidar_data:
            # Find minimum distance in front (±30 degrees)
            center_idx = len(self.lidar_data.ranges) // 2
            scan_width = len(self.lidar_data.ranges) // 12  # ~±30 degrees

            min_dist = float('inf')
            for i in range(max(0, center_idx - scan_width), min(len(self.lidar_data.ranges), center_idx + scan_width)):
                if not math.isnan(self.lidar_data.ranges[i]) and self.lidar_data.ranges[i] < min_dist:
                    min_dist = self.lidar_data.ranges[i]

            self.obstacle_distance = min_dist
            self.obstacle_detected = min_dist < 2.0  # Obstacle within 2 meters

            if self.obstacle_detected:
                self.get_logger().warn(f'Obstacle detected at {min_dist:.2f}m')

    def imu_callback(self, msg):
        """
        Handle incoming IMU data
        """
        self.imu_data = msg

    def joint_callback(self, msg):
        """
        Handle incoming joint state data
        """
        self.joint_data = msg

    def control_loop(self):
        """
        Main control loop - runs at specified frequency
        """
        # Check for sensor timeouts
        if time.time() - self.last_sensor_time > self.safety_timeout:
            self.get_logger().error('Sensor data timeout - entering safe state')
            self.system_state = 'FAULT'
            self.stop_motors()
            return

        if self.system_state != 'ACTIVE':
            return  # Don't run control if not active

        # Select control algorithm based on parameter
        if self.control_algorithm == 'obstacle_avoidance':
            cmd_vel = self.obstacle_avoidance_control()
        elif self.control_algorithm == 'waypoint_following':
            cmd_vel = self.waypoint_following_control()
        else:
            cmd_vel = self.default_control()

        # Publish motor commands
        if cmd_vel:
            self.cmd_pub.publish(cmd_vel)

            # Also publish individual motor commands
            motor1_cmd = Float64()
            motor1_cmd.data = cmd_vel.linear.x  # Simplified mapping
            self.motor1_pub.publish(motor1_cmd)

            motor2_cmd = Float64()
            motor2_cmd.data = cmd_vel.angular.z  # Simplified mapping
            self.motor2_pub.publish(motor2_cmd)

    def obstacle_avoidance_control(self):
        """
        Obstacle avoidance control algorithm
        """
        cmd_vel = Twist()

        if self.obstacle_detected and self.obstacle_distance < 1.0:
            # Emergency stop if obstacle too close
            cmd_vel.linear.x = 0.0
            cmd_vel.angular.z = 0.0
            self.get_logger().warn('Emergency stop: obstacle too close!')
        elif self.obstacle_detected and self.obstacle_distance < 2.0:
            # Slow down and turn away
            cmd_vel.linear.x = max(0.0, self.max_lin_vel * (self.obstacle_distance - 0.5))
            if self.lidar_data:
                center_idx = len(self.lidar_data.ranges) // 2
                left_avg = sum(self.lidar_data.ranges[:center_idx]) / len(self.lidar_data.ranges[:center_idx])
                right_avg = sum(self.lidar_data.ranges[center_idx:]) / len(self.lidar_data.ranges[center_idx:])

                # Turn away from closer obstacle
                if left_avg < right_avg:
                    cmd_vel.angular.z = -self.max_ang_vel * 0.5  # Turn right
                else:
                    cmd_vel.angular.z = self.max_ang_vel * 0.5   # Turn left
        else:
            # Move forward normally
            cmd_vel.linear.x = self.max_lin_vel
            cmd_vel.angular.z = 0.0

        # Apply velocity limits
        cmd_vel.linear.x = max(-self.max_lin_vel, min(self.max_lin_vel, cmd_vel.linear.x))
        cmd_vel.angular.z = max(-self.max_ang_vel, min(self.max_ang_vel, cmd_vel.angular.z))

        return cmd_vel

    def waypoint_following_control(self):
        """
        Waypoint following control algorithm (simplified)
        """
        # This would typically involve path planning and following
        # For now, just implement basic forward movement
        cmd_vel = Twist()
        cmd_vel.linear.x = self.max_lin_vel * 0.5
        cmd_vel.angular.z = 0.0
        return cmd_vel

    def default_control(self):
        """
        Default control algorithm
        """
        cmd_vel = Twist()
        cmd_vel.linear.x = 0.0
        cmd_vel.angular.z = 0.0
        return cmd_vel

    def emergency_stop_callback(self, request, response):
        """
        Handle emergency stop service call
        """
        self.get_logger().warn('EMERGENCY STOP REQUESTED')
        self.system_state = 'EMERGENCY_STOPPED'
        self.stop_motors()
        response.success = True
        response.message = 'Emergency stop activated'
        return response

    def reset_controller_callback(self, request, response):
        """
        Handle controller reset service call
        """
        self.get_logger().info('CONTROLLER RESET REQUESTED')
        self.system_state = 'ACTIVE'
        response.success = True
        response.message = 'Controller reset to active state'
        return response

    def stop_motors(self):
        """
        Stop all motors immediately
        """
        cmd_vel = Twist()
        cmd_vel.linear.x = 0.0
        cmd_vel.angular.z = 0.0

        self.cmd_pub.publish(cmd_vel)

        motor1_cmd = Float64()
        motor1_cmd.data = 0.0
        self.motor1_pub.publish(motor1_cmd)

        motor2_cmd = Float64()
        motor2_cmd.data = 0.0
        self.motor2_pub.publish(motor2_cmd)

    def on_shutdown(self):
        """
        Cleanup function called on shutdown
        """
        self.get_logger().info('Shutting down controller node')
        self.stop_motors()


def main(args=None):
    rclpy.init(args=args)

    controller_node = ControllerNode()

    try:
        rclpy.spin(controller_node)
    except KeyboardInterrupt:
        controller_node.get_logger().info('Keyboard interrupt received')
    finally:
        controller_node.on_shutdown()
        controller_node.destroy_node()
        rclpy.shutdown()


if __name__ == '__main__':
    main()
```

### 1.3 Actuator Node Implementation

Now, let's implement a basic actuator node:

```python
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import Twist
from std_msgs.msg import Float64, String
from std_srvs.srv import Trigger
import time


class ActuatorNode(Node):
    """
    Actuator node that receives motor commands and simulates motor responses
    """
    def __init__(self, node_name='actuator_node', motor_id=1):
        super().__init__(node_name)

        # Declare parameters as per specification
        self.declare_parameter('max_velocity', 1.0)
        self.declare_parameter('max_torque', 10.0)
        self.declare_parameter('deadband', 0.01)
        self.declare_parameter('motor_type', 'continuous_rotation')

        # Get parameter values
        self.max_velocity = self.get_parameter('max_velocity').value
        self.max_torque = self.get_parameter('max_torque').value
        self.deadband = self.get_parameter('deadband').value
        self.motor_type = self.get_parameter('motor_type').value

        self.motor_id = motor_id

        # Create subscribers for motor commands
        if motor_id == 1:
            self.cmd_sub = self.create_subscription(
                Twist, 'motor1_command', self.motor1_cmd_callback, 10)
        elif motor_id == 2:
            self.cmd_sub = self.create_subscription(
                Twist, 'motor2_command', self.motor2_cmd_callback, 10)
        else:
            self.cmd_sub = self.create_subscription(
                Float64, f'motor{motor_id}_command', self.generic_cmd_callback, 10)

        # Create publishers for motor status
        self.status_pub = self.create_publisher(
            Float64, f'motor{motor_id}_status', 10)

        # Create timer for status updates
        self.status_timer = self.create_timer(0.1, self.publish_status)

        # Motor state variables
        self.current_velocity = 0.0
        self.target_velocity = 0.0
        self.is_active = True
        self.error_state = False
        self.last_cmd_time = time.time()

        self.get_logger().info(f'{node_name} initialized for motor {motor_id}')

    def motor1_cmd_callback(self, msg):
        """
        Handle Twist command for differential drive
        """
        # For motor 1, use linear x component
        self.target_velocity = max(-self.max_velocity, min(self.max_velocity, msg.linear.x))
        self.last_cmd_time = time.time()

        # Apply deadband
        if abs(self.target_velocity) < self.deadband:
            self.target_velocity = 0.0

        self.get_logger().debug(f'Motor {self.motor_id} target velocity: {self.target_velocity}')

    def motor2_cmd_callback(self, msg):
        """
        Handle Twist command for differential drive (second motor)
        """
        # For motor 2, use angular z component
        self.target_velocity = max(-self.max_velocity, min(self.max_velocity, msg.angular.z))
        self.last_cmd_time = time.time()

        # Apply deadband
        if abs(self.target_velocity) < self.deadband:
            self.target_velocity = 0.0

        self.get_logger().debug(f'Motor {self.motor_id} target velocity: {self.target_velocity}')

    def generic_cmd_callback(self, msg):
        """
        Handle generic Float64 command
        """
        self.target_velocity = max(-self.max_velocity, min(self.max_velocity, msg.data))
        self.last_cmd_time = time.time()

        # Apply deadband
        if abs(self.target_velocity) < self.deadband:
            self.target_velocity = 0.0

        self.get_logger().debug(f'Motor {self.motor_id} target velocity: {self.target_velocity}')

    def publish_status(self):
        """
        Publish motor status information
        """
        # Check for command timeouts
        if time.time() - self.last_cmd_time > 1.0:  # 1 second timeout
            self.get_logger().warn(f'Motor {self.motor_id}: Command timeout - stopping motor')
            self.target_velocity = 0.0

        # Simulate motor dynamics
        if abs(self.target_velocity - self.current_velocity) > 0.01:
            # Simple first-order dynamics
            dt = 0.1  # Timer period
            tau = 0.2  # Time constant
            self.current_velocity += (self.target_velocity - self.current_velocity) * dt / tau

        # Create and publish status message
        status_msg = Float64()
        status_msg.data = self.current_velocity
        self.status_pub.publish(status_msg)

    def on_shutdown(self):
        """
        Cleanup function called on shutdown
        """
        self.get_logger().info(f'Shutting down actuator node for motor {self.motor_id}')
        # Stop motor
        self.target_velocity = 0.0
        self.current_velocity = 0.0
        # Publish final zero status
        status_msg = Float64()
        status_msg.data = 0.0
        self.status_pub.publish(status_msg)


def main(args=None):
    rclpy.init(args=args)

    # Create multiple actuator nodes
    actuator1 = ActuatorNode('actuator1_node', 1)
    actuator2 = ActuatorNode('actuator2_node', 2)
    actuator3 = ActuatorNode('actuator3_node', 3)

    try:
        executor = rclpy.executors.MultiThreadedExecutor()
        executor.add_node(actuator1)
        executor.add_node(actuator2)
        executor.add_node(actuator3)

        executor.spin()
    except KeyboardInterrupt:
        pass
    finally:
        actuator1.on_shutdown()
        actuator2.on_shutdown()
        actuator3.on_shutdown()

        actuator1.destroy_node()
        actuator2.destroy_node()
        actuator3.destroy_node()
        rclpy.shutdown()


if __name__ == '__main__':
    main()
```

## Phase 3: Safety Features and Error Handling

Let's enhance the controller node with additional safety features:

```python
# Enhanced safety features for the controller node
class EnhancedControllerNode(ControllerNode):
    """
    Enhanced controller node with additional safety features
    """
    def __init__(self, node_name='enhanced_controller_node'):
        super().__init__(node_name)

        # Additional safety parameters
        self.declare_parameter('watchdog_timeout', 0.5)
        self.declare_parameter('velocity_ramp_rate', 2.0)  # m/s^2
        self.declare_parameter('position_limits_enabled', True)
        self.declare_parameter('position_min', -10.0)
        self.declare_parameter('position_max', 10.0)

        # Get additional parameters
        self.watchdog_timeout = self.get_parameter('watchdog_timeout').value
        self.velocity_ramp_rate = self.get_parameter('velocity_ramp_rate').value
        self.pos_limits_enabled = self.get_parameter('position_limits_enabled').value
        self.pos_min = self.get_parameter('position_min').value
        self.pos_max = self.get_parameter('position_max').value

        # Initialize safety state variables
        self.last_cmd_timestamp = self.get_clock().now()
        self.current_cmd_vel = Twist()
        self.ramped_cmd_vel = Twist()
        self.position_x = 0.0
        self.position_y = 0.0

        # Create timer for watchdog
        self.watchdog_timer = self.create_timer(
            self.watchdog_timeout, self.watchdog_check)

        # Create diagnostic publisher
        self.diag_pub = self.create_publisher(
            String, 'system_diagnostics', 10)

        self.get_logger().info(f'{node_name} initialized with enhanced safety features')

    def control_loop(self):
        """
        Enhanced control loop with safety features
        """
        # Check for sensor timeouts
        if time.time() - self.last_sensor_time > self.safety_timeout:
            self.get_logger().error('Sensor data timeout - entering safe state')
            self.system_state = 'FAULT'
            self.stop_motors()
            return

        if self.system_state != 'ACTIVE':
            return  # Don't run control if not active

        # Update last command timestamp
        self.last_cmd_timestamp = self.get_clock().now()

        # Select control algorithm based on parameter
        if self.control_algorithm == 'obstacle_avoidance':
            cmd_vel = self.obstacle_avoidance_control()
        elif self.control_algorithm == 'waypoint_following':
            cmd_vel = self.waypoint_following_control()
        else:
            cmd_vel = self.default_control()

        # Apply safety limits and ramping
        cmd_vel = self.apply_safety_limits(cmd_vel)
        cmd_vel = self.apply_velocity_ramping(cmd_vel)

        # Update position estimates
        self.update_position_estimate(cmd_vel)

        # Check position limits
        if self.pos_limits_enabled:
            if self.position_x < self.pos_min or self.position_x > self.pos_max:
                self.get_logger().warn(f'Position limit exceeded: {self.position_x}')
                cmd_vel.linear.x = 0.0
                cmd_vel.angular.z = 0.0

        # Store current command
        self.current_cmd_vel = cmd_vel

        # Publish motor commands
        if cmd_vel:
            self.cmd_pub.publish(cmd_vel)

            # Also publish individual motor commands
            motor1_cmd = Float64()
            motor1_cmd.data = cmd_vel.linear.x  # Simplified mapping
            self.motor1_pub.publish(motor1_cmd)

            motor2_cmd = Float64()
            motor2_cmd.data = cmd_vel.angular.z  # Simplified mapping
            self.motor2_pub.publish(motor2_cmd)

    def apply_safety_limits(self, cmd_vel):
        """
        Apply safety limits to command velocities
        """
        # Apply velocity limits
        cmd_vel.linear.x = max(-self.max_lin_vel, min(self.max_lin_vel, cmd_vel.linear.x))
        cmd_vel.angular.z = max(-self.max_ang_vel, min(self.max_ang_vel, cmd_vel.angular.z))

        # Apply acceleration limits through ramping (handled in apply_velocity_ramping)

        return cmd_vel

    def apply_velocity_ramping(self, cmd_vel):
        """
        Apply velocity ramping to prevent sudden changes
        """
        # Calculate maximum change allowed based on ramp rate and time delta
        current_time = self.get_clock().now()
        time_delta = (current_time - self.last_cmd_timestamp).nanoseconds / 1e9  # Convert to seconds

        if time_delta > 0.1:  # If more than 100ms has passed, reset ramping
            self.ramped_cmd_vel = self.current_cmd_vel

        max_change = self.velocity_ramp_rate * time_delta

        # Ramp linear velocity
        lin_diff = cmd_vel.linear.x - self.ramped_cmd_vel.linear.x
        if abs(lin_diff) > max_change:
            sign = 1 if lin_diff > 0 else -1
            cmd_vel.linear.x = self.ramped_cmd_vel.linear.x + sign * max_change

        # Ramp angular velocity
        ang_diff = cmd_vel.angular.z - self.ramped_cmd_vel.angular.z
        if abs(ang_diff) > max_change:
            sign = 1 if ang_diff > 0 else -1
            cmd_vel.angular.z = self.ramped_cmd_vel.angular.z + sign * max_change

        # Update ramped command
        self.ramped_cmd_vel = cmd_vel

        return cmd_vel

    def update_position_estimate(self, cmd_vel):
        """
        Update position estimate based on commanded velocities
        """
        dt = 1.0 / self.control_freq
        self.position_x += cmd_vel.linear.x * dt * math.cos(self.position_y)  # Simplified
        self.position_y += cmd_vel.linear.x * dt * math.sin(self.position_y)  # Simplified

    def watchdog_check(self):
        """
        Watchdog timer to check for stuck conditions
        """
        current_time = self.get_clock().now()
        time_since_last_cmd = (current_time - self.last_cmd_timestamp).nanoseconds / 1e9

        if time_since_last_cmd > self.watchdog_timeout * 2:  # Double timeout for safety
            self.get_logger().error('Watchdog timeout - no commands received, stopping motors')
            self.stop_motors()

            # Publish diagnostic message
            diag_msg = String()
            diag_msg.data = f'WATCHDOG_TIMEOUT: No commands for {time_since_last_cmd:.2f}s'
            self.diag_pub.publish(diag_msg)

    def on_shutdown(self):
        """
        Enhanced cleanup function called on shutdown
        """
        self.get_logger().info('Shutting down enhanced controller node')
        self.stop_motors()

        # Publish final diagnostic
        diag_msg = String()
        diag_msg.data = 'SYSTEM_SHUTDOWN: Controller node shutting down safely'
        self.diag_pub.publish(diag_msg)
```

```python
        lidar_msg.ranges = [random.uniform(1.0, 5.0) for _ in range(181)]

        self.start_time = time.time()
        self.lidar_pub.publish(lidar_msg)

        Print performance report

        if len(self.latency_samples) > 0:
            avg_latency = statistics.mean(self.latency_samples)
            min_latency = min(self.latency_samples)
            max_latency = max(self.latency_samples)
            std_dev = statistics.stdev(self.latency_samples) if len(self.latency_samples) > 1 else 0.0

            self.get_logger().info(f'Max Latency: {max_latency*1000:.2f} ms')
            self.get_logger().info(f'Std Dev: {std_dev*1000:.2f} ms')
            self.get_logger().info(f'Throughput: {self.message_count/2:.2f} Hz (approx)')
            self.get_logger().info('=========================')
```
